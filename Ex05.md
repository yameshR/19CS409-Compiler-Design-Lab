## PROGRAM

#include <stdio.h> 
/* This YACC program is for recognizing the Expression */ 
%} 
%token A ID 
%% 
statement: A'='E 
| E { 
printf("\n Valid arithmetic expression"); 
$$=$1; 
} 
; 
  E: E'+'ID 
| E'-'ID 
| E'*'ID 
| E'/'ID 
| ID 
; 
%%

yyparse(); 
}while(!feof(yyin)); }

}
